        // Тіл ауысқанда немесе парақша жүктелгенде интерфейсті жаңарту
        function updateLanguageAndContent() {
            const lang = document.getElementById('langSelect').value;
            
            // Көптілді мәтіндерді жаңарту
            document.querySelectorAll('[data-i18n]').forEach(el => {
                const key = el.dataset.i18n;
                if (i18n[lang][key]) el.innerText = i18n[lang][key];
            });

            // Іздеу жолағының мәтінін жаңарту
            const searchInput = document.getElementById('searchInput');
            if (searchInput) {
                searchInput.placeholder = i18n[lang].search;
            }

            // Егер қазір бір белсенді бөлім ашық болса, оның ішін жаңарту
            if (currentActiveCategoryKey) {
                openCategory(currentActiveCategoryKey, true);
            }
        }

        // Алфавит әріптерін экранға шығару
        function renderAlphabet(lang) {
            const grid = document.getElementById('alphabetGrid');
            grid.innerHTML = '';
            
            alphabets[lang].forEach(letter => {
                const div = document.createElement('div');
                div.className = 'letter-box';
                div.innerText = letter;
                div.id = `letter-${letter}`;
                div.onclick = () => selectLetter(letter);
                grid.appendChild(div);
            });
        }

        // Әріпті таңдаған кезде оған қатысты видео аймағын ашу
        function selectLetter(letter) {
            document.querySelectorAll('.letter-box').forEach(el => {
                el.classList.remove('selected');
            });

            const selectedBox = document.getElementById(`letter-${letter}`);
            if (selectedBox) {
                selectedBox.classList.add('selected');
            }

            const videoContainer = document.getElementById('videoContainer');
            videoContainer.classList.remove('hidden');
            
            const currentLetterDisplay = document.getElementById('currentLetterDisplay');
            currentLetterDisplay.innerText = letter;

            document.getElementById('alphabetPage').scrollTo({ top: 0, behavior: 'smooth' });
        }

        // Кез келген сөзді таңдағанда оның сипаттамасы мен қимылын плеерде көрсету
        function selectWord(wordObj, elementId) {
            document.querySelectorAll('.word-item').forEach(el => {
                el.classList.remove('selected', 'bg-emerald-600', 'text-white');
                el.classList.add('bg-slate-50', 'text-slate-800');
            });

            const selectedBtn = document.getElementById(elementId);
            if (selectedBtn) {
                selectedBtn.classList.remove('bg-slate-50', 'text-slate-800');
                selectedBtn.classList.add('selected', 'bg-emerald-600', 'text-white');
            }

            // Плеер мазмұнын жаңарту
            document.getElementById('currentWordDisplay').innerText = wordObj.word;
            document.getElementById('currentWordDesc').innerText = wordObj.desc;

            // Фокусты жоғары жылжыту
            document.getElementById('generalCategoryPage').scrollTo({ top: 0, behavior: 'smooth' });
        }

        // Кез келген бөлімді ашқан кездегі бағыттау функциясы
        function openCategory(key, isLanguageRefresh = false) {
            currentActiveCategoryKey = key;
            const lang = document.getElementById('langSelect').value;

            // Барлық беттерді жабу
            document.querySelectorAll('.page').forEach(page => page.classList.remove('active'));

            if (key === 'cat1') {
                // Алфавит бетін ашу
                document.getElementById('alphabetPage').classList.add('active');
                document.getElementById('categoryTitle').innerText = i18n[lang][key];
                renderAlphabet(lang);
                // Әдепкі бойынша бірінші әріпті таңдау
                selectLetter(alphabets[lang][0]);
            } else {
                // Қалған 5 негізгі бөлімдерді (Сұрақтар, Диалог, Әрекеттер, Отбасы, Глухой) ашу
                document.getElementById('generalCategoryPage').classList.add('active');
                document.getElementById('generalTitle').innerText = i18n[lang][key];
                document.getElementById('currentWordCategory').innerText = i18n[lang][key];

                // Осы категорияның сөздерін тізімге қосу
                const listContainer = document.getElementById('wordsListContainer');
                listContainer.innerHTML = '';

                const words = categoryData[key][lang];
                words.forEach((item, index) => {
                    const btnId = `word-item-${index}`;
                    const btn = document.createElement('button');
                    btn.id = btnId;
                    btn.className = 'word-item w-full p-4 rounded-2xl bg-slate-50 border border-slate-100 flex justify-between items-center text-left font-bold text-slate-800 shadow-sm';
                    btn.innerHTML = `
                        <span>${item.word}</span>
                        <span class="text-emerald-500 text-sm">▶</span>
                    `;
                    btn.onclick = () => selectWord(item, btnId);
                    listContainer.appendChild(btn);
                });

                // Әдепкі бойынша бірінші сөзді таңдап көрсету
                if (words.length > 0) {
                    selectWord(words[0], `word-item-0`);
                }
            }
        }

        function showHome() {
            currentActiveCategoryKey = "";
            document.querySelectorAll('.page').forEach(page => page.classList.remove('active'));
            document.getElementById('homePage').classList.add('active');
            
            // Іздеу жолағын тазалау
            document.getElementById('searchInput').value = '';
        }

        // Бөлімдерді ашу (Басты бетке өтіп, категориялар секциясына бағыттайды)
        function goToCategories() {
            showHome();
            const categoriesSec = document.getElementById('categoriesSection');
            if (categoriesSec) {
                categoriesSec.scrollIntoView({ behavior: 'smooth' });
            }
        }

        // Профиль (Жеке кабинет) бетін ашу

            currentActiveCategoryKey = "";
            document.querySelectorAll('.page').forEach(page => page.classList.remove('active'));
            document.getElementById('profilePage').classList.add('active');
        }

        // Іздеу функциясы
        function handleSearch() {
            const query = document.getElementById('searchInput').value.toLowerCase().trim();
            const lang = document.getElementById('langSelect').value;
            
            if (query.length > 0) {
                // Алдымен әріптерден іздеу
                const matchedLetter = alphabets[lang].find(l => l.toLowerCase() === query);
                if (matchedLetter) {
                    openCategory('cat1');
                    selectLetter(matchedLetter);
                    return;
                }

                // Екінші кезекте барлық басқа бөлімдердегі сөздерден іздеу
                for (const catKey in categoryData) {
                    const matchedWord = categoryData[catKey][lang].find(item => item.word.toLowerCase().includes(query));
                    if (matchedWord) {
                        openCategory(catKey);
                        // Осы сөзді тізімнен тауып таңдау
                        const words = categoryData[catKey][lang];
                        const wordIndex = words.findIndex(w => w.word === matchedWord.word);
                        if (wordIndex !== -1) {
                            selectWord(matchedWord, `word-item-${wordIndex}`);
                        }
                        return;
                    }
                }
            }
        }
